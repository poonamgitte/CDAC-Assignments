Examples: 
/* No transformation */
transform: none;

/* 2D translation: Move right by 50px and down by 30px */
transform: translate(50px, 30px);

/* Scale up twice as wide and tall */
transform: scale(2, 2);

/* Rotate by 45 degrees */
transform: rotate(45deg);

/* Skew X by 20deg and Y by 10deg */
transform: skew(20deg, 10deg);

/* Apply transformation matrix */
transform: matrix(1, 0.5, -0.5, 1, 30, 20);

/* Set origin to left edge (x-axis) */
transform-origin: 0;

/* Center vertically (y-axis) */
transform-origin: 50% 100%;

/* Use z-axis for 3D */
transform-origin: 50% 50% 20px;

/* Use pixel values */
transform-origin: 20px 40px;

/* Default (no perspective) */
perspective: none;

/* Giving a 3D perspective of 800px */
perspective: 800px;

/* Show the "back" of an element */
backface-visibility: visible;

/* Hide the back */
backface-visibility: hidden;

/* No filter */
filter: none;

/* Blur effect */
filter: blur(4px);

/* Increase brightness */
filter: brightness(150%);

/* Increase contrast */
filter: contrast(2);

/* Drop shadow effect */
filter: drop-shadow(2px 4px 6px black);

/* Grayscale effect */
filter: grayscale(70%);

/* Hue rotation */
filter: hue-rotate(120deg);

/* Invert colors */
filter: invert(1);

/* No shadow */
text-shadow: none;

/* Horizontal 2px, vertical 5px, blur 10px, black */
text-shadow: 2px 5px 10px #000;

/* Multiple shadows */
text-shadow: 1px 1px 2px gray, 0 0 1em blue;

/* No clip */
clip-path: none;

/* Polygon shape */
clip-path: polygon(50% 0%, 100% 50%, 50% 100%, 0% 50%);

/* Circular shape */
clip-path: circle(40% at 60% 40%);

/* Ellipse shape */
clip-path: ellipse(60% 40% at 40% 60%);

/* Inset (rectangular) */
clip-path: inset(5% 10% 15% 20%);

/* No mask */
mask-image: none;

/* Mask using an image URL */
mask-image: url('sample.png');

/* Default: auto (element doesn't isolate) */
isolation: auto;

/* Force element into its own stacking context */
isolation: isolate;
